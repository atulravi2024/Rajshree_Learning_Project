#### 2. Quiz Logic (`ui/quiz.js`) — [NEW FILE]
Create this module to handle the random selection of 4 options and 1 correct answer.
```javascript
function startQuiz() {
    window.isQuizActive = true;
    window.quizScore = 0;
    playSound('system/quiz/quiz_start.mp3');
    generateNewQuestion();
}

function generateNewQuestion() {
    const data = window.activeCardsData;
    const correct = data[Math.floor(Math.random() * data.length)];
    const distractors = data.filter(i => i.word !== correct.word)
                            .sort(() => 0.5 - Math.random())
                            .slice(0, 3);
    window.quizQuestion = correct;
    window.quizOptions = [correct, ...distractors].sort(() => 0.5 - Math.random());
    renderQuizUI();
}
```

## 💾 Phase 2: Local Progress & Persistence
**Goal:** Make the app remember user progress (categories completed, stars earned) using `localStorage`.

### 📋 Deep Technical Blueprint

#### 1. Persistence Logic (`utils/helpers.js`)
Add these specialized methods:
```javascript
const PROGRESS_KEY = 'RAJSHREE_APP_V1_PROGRESS';

function saveCategoryProgress(catKey, isComplete) {
    let progress = JSON.parse(localStorage.getItem(PROGRESS_KEY)) || {};
    progress[catKey] = {
        lastVisited: new Date().toISOString(),
        completed: isComplete || progress[catKey]?.completed || false,
        stars: isComplete ? 3 : (progress[catKey]?.stars || 0)
    };
    localStorage.setItem(PROGRESS_KEY, JSON.stringify(progress));
}

function loadProgress() {
    return JSON.parse(localStorage.getItem(PROGRESS_KEY)) || {};
}
```
